暴力递归出发分析动态规划

public int maxProfit1(int k, int[] prices) {
    // 暴力
    if(prices.length== 0 || prices == null){
        return 0;
    }
    return dfs(0, 0, 0, k, prices);
}

public int dfs(int index, int status, int count, int k, int[] prices){
    if (count == k || index == prices.length) {
        return 0;
    }
    // 保持不变
    int a = 0, b = 0, c = 0;
    a = dfs(index + 1, status, count, k, prices);
    if (status == 1){
        // 卖出
        b = dfs(index + 1, 0, count + 1, k, prices) + prices[index];
    }else {
        // 买入
        c = dfs(index + 1, 1, count, k, prices) - prices[index];
    }
    return Math.max(Math.max(a, b), c);
}

动态规划

三个参数三维表，但有一维为01，转为两个二维表

// 动态规划
public int maxProfit2(int k, int[] prices) {
    if(prices.length== 0 || prices == null){
        return 0;
    }
    int n = prices.length;
    k = Math.min(n/2, k);
    // 动态规划的意义
    // buy[i][j]表示到达索引i时候，达成j次交易买入获得最大收益
    // sell[i][j]表示到达索引i时候，达成j次交易卖出获得最大收益
    // buy[i][j] = Math.max( buy[i - 1][j], sell[i - 1][j] - price[i])
    // sell[i][j] = Math.max( sell[i - 1][j], buy[i - 1][j - 1] + price[i])
    int[][] buy = new int[n][k + 1];
    int[][] sell = new int[n][k + 1];
    buy[0][0] = -prices[0];
    sell[0][0] = 0;

    // 注意：buy[0][i] 和 sell[0][i] 是没有意义的
    for (int i = 1; i <= k; ++i) {
        buy[0][i] = sell[0][i] = Integer.MIN_VALUE / 2;
    }

    for (int i = 1; i < n; i++){
        buy[i][0] = Math.max( buy[i - 1][0], sell[i - 1][0] - prices[i]);
        for (int j = 1; j<= k; j++){
            buy[i][j] = Math.max( buy[i - 1][j], sell[i - 1][j] - prices[i]);
            sell[i][j] = Math.max( sell[i - 1][j], buy[i - 1][j - 1] + prices[i]);
        }
    }

    return Arrays.stream(sell[n - 1]).max().getAsInt();
}

动态规划优化

不难看出上述二维表中的i并没有被实际用到

public int maxProfit(int k, int[] prices) {
    // 动态规划空间优化
    if(prices.length== 0 || prices == null){
        return 0;
    }
    int n = prices.length;
    k = Math.min(n/2, k);

    int[] buy = new int[k + 1];
    int[] sell = new int[k + 1];
    buy[0] = -prices[0];
    sell[0] = 0;

    // 注意：buy[0][i] 和 sell[0][i] 是没有意义的
    for (int i = 1; i <= k; ++i) {
        buy[i] = sell[i] = Integer.MIN_VALUE / 2;
    }

    for (int i = 1; i < n; i++){
        buy[0] = Math.max( buy[0], sell[0] - prices[i]);
        for (int j = 1; j<= k; j++){
            buy[j] = Math.max( buy[j], sell[j] - prices[i]);
            sell[j] = Math.max( sell[j], buy[j - 1] + prices[i]);
        }
    }

    return Arrays.stream(sell).max().getAsInt();
}
